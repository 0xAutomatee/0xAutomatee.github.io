Chapter 4

SELECT & FROM

The foundation of every SQL query. SELECT specifies the columns to return; FROM specifies the source table.

Execution Order The computer executes clauses in this order: FROM → WHERE → GROUP BY → HAVING → SELECT → ORDER BY — not the order you write them.
1FROM
2WHERE
3GROUP BY
4HAVING
5SELECT
6ORDER BY

Basic SELECT

SQL
basic_select.sql
-- Select all columns from a table
SELECT *
FROM orders;

-- Select specific columns
SELECT order_id, customer_name, order_total
FROM orders;

-- Column alias with AS
SELECT
    order_id                    AS "Order #",
    customer_name               AS customer,
    order_total * 1.1           AS total_with_tax
FROM orders;

DISTINCT — Remove Duplicates

SQL
distinct.sql
-- Return unique values only
SELECT DISTINCT retailer
FROM orders;

-- DISTINCT across multiple columns
SELECT DISTINCT retailer, status
FROM orders;

Subquery in SELECT

SQL
select_subquery.sql
-- Embed a subquery as a derived column
SELECT
    order_id,
    order_total,
    (SELECT AVG(order_total) FROM orders) AS avg_order
FROM orders;
Chapter 4

WHERE Clause

Filter rows before grouping or aggregating. Supports comparison, logical, and pattern-matching operators.

Comparison & Logical Operators

SQL
where_operators.sql
SELECT * FROM orders
WHERE order_total > 500;

-- AND / OR / NOT
SELECT * FROM orders
WHERE retailer = 'QVC'
  AND status != 'cancelled'
  AND order_total BETWEEN 100 AND 1000;

-- IN list
SELECT * FROM orders
WHERE retailer IN ('QVC', 'HSN', 'Amazon');

-- NULL checks
SELECT * FROM payments
WHERE matched_invoice IS NULL;

LIKE — Pattern Matching

SQL
like_patterns.sql
-- % matches any sequence of characters
SELECT * FROM invoices
WHERE invoice_num LIKE 'INV-%';

-- _ matches exactly one character
SELECT * FROM accounts
WHERE account_code LIKE 'HA____';

-- NOT LIKE
SELECT * FROM orders
WHERE product_name NOT LIKE '%sample%';

Subquery in WHERE

SQL
where_subquery.sql
-- Filter using a subquery result
SELECT * FROM orders
WHERE customer_id IN (
    SELECT customer_id
    FROM customers
    WHERE tier = 'premium'
);

-- EXISTS check
SELECT * FROM invoices i
WHERE EXISTS (
    SELECT 1 FROM payments p
    WHERE p.invoice_id = i.invoice_id
);
Chapter 4

GROUP BY & HAVING

GROUP BY splits rows into groups for aggregation. HAVING filters those groups — it's WHERE for aggregated results.

Key Rule Every column in SELECT must either be in GROUP BY or wrapped in an aggregate function like SUM(), COUNT(), AVG().

GROUP BY Basics

SQL
group_by.sql
-- Total sales per retailer
SELECT
    retailer,
    COUNT(*)          AS order_count,
    SUM(order_total)  AS total_sales,
    AVG(order_total)  AS avg_order
FROM orders
GROUP BY retailer;

-- Group by multiple columns
SELECT
    retailer,
    status,
    COUNT(*) AS cnt
FROM orders
GROUP BY retailer, status;

HAVING — Filter Groups

SQL
having.sql
-- Only retailers with over $10,000 in sales
SELECT
    retailer,
    SUM(order_total) AS total_sales
FROM orders
GROUP BY retailer
HAVING SUM(order_total) > 10000;

-- WHERE filters rows; HAVING filters groups
SELECT retailer, COUNT(*) AS cnt
FROM orders
WHERE  status = 'shipped'          -- row-level filter
GROUP BY retailer
HAVING COUNT(*) >= 5;              -- group-level filter
Chapter 4

ORDER BY & LIMIT

Sort and paginate your result set. LIMIT (or TOP in SQL Server) caps the number of rows returned.

SQL
order_limit.sql
-- Sort ascending (default) and descending
SELECT order_id, order_total
FROM orders
ORDER BY order_total DESC;

-- Sort by multiple columns
SELECT * FROM orders
ORDER BY retailer ASC, order_date DESC;

-- MySQL / PostgreSQL / SQLite: LIMIT
SELECT * FROM orders
ORDER BY order_date DESC
LIMIT 10;

-- SQL Server: TOP
SELECT TOP 10 * FROM orders
ORDER BY order_date DESC;

-- Oracle: ROWNUM / FETCH
SELECT * FROM orders
ORDER BY order_date DESC
FETCH FIRST 10 ROWS ONLY;

CRUD Operations

Create, Read, Update, Delete — the four fundamental operations on tables and data.

CREATE TABLE

SQL
create_table.sql
CREATE TABLE orders (
    order_id      INT           PRIMARY KEY,
    retailer      VARCHAR(50)   NOT NULL,
    order_date    DATE          NOT NULL,
    order_total   DECIMAL(10,2) DEFAULT 0.00,
    status        VARCHAR(20)   DEFAULT 'pending',
    customer_id   INT           REFERENCES customers(customer_id)
);

-- Create only if it doesn't exist
CREATE TABLE IF NOT EXISTS payments (
    payment_id    INT           PRIMARY KEY,
    amount        DECIMAL(10,2),
    paid_date     DATE
);

INSERT, UPDATE, DELETE

SQL
dml.sql
-- INSERT a row
INSERT INTO orders (order_id, retailer, order_date, order_total)
VALUES (1001, 'QVC', '2024-01-15', 249.99);

-- INSERT from a SELECT
INSERT INTO archive_orders
SELECT * FROM orders
WHERE order_date < '2023-01-01';

-- UPDATE rows
UPDATE orders
SET    status = 'paid',
       updated_at = CURRENT_TIMESTAMP
WHERE  order_id = 1001;

-- DELETE rows
DELETE FROM orders
WHERE status = 'cancelled'
  AND order_date < '2022-01-01';

-- DROP entire table
DROP TABLE IF EXISTS temp_orders;

Aggregate Functions

Compute summary statistics across groups of rows. Used with GROUP BY or as window functions.

SQL
aggregates.sql
SELECT
    COUNT(*)                   AS total_rows,
    COUNT(payment_id)          AS non_null_payments,
    COUNT(DISTINCT retailer)  AS unique_retailers,
    SUM(order_total)           AS grand_total,
    AVG(order_total)           AS average_order,
    MIN(order_total)           AS smallest_order,
    MAX(order_total)           AS largest_order
FROM orders;
Chapter 7

String Functions

Manipulate and format text data. Syntax varies slightly between RDBMS platforms.

SQL
string_functions.sql
SELECT
    -- Length
    LEN(product_name)                    AS name_length,   -- SQL Server
    LENGTH(product_name)                 AS name_length,   -- MySQL/PG/SQLite

    -- Case
    UPPER(retailer)                      AS retailer_upper,
    LOWER(retailer)                      AS retailer_lower,

    -- Trim whitespace
    TRIM(customer_name)                  AS clean_name,
    LTRIM(RTRIM(customer_name))          AS clean_name_2,  -- SQL Server

    -- Concatenate
    first_name || ' ' || last_name       AS full_name,    -- PG/SQLite/Oracle
    CONCAT(first_name, ' ', last_name)   AS full_name_2,  -- MySQL/SQL Server

    -- Substring
    SUBSTRING(invoice_num, 1, 3)          AS prefix,

    -- Replace
    REPLACE(phone, '-', '')              AS phone_clean

FROM orders;
Chapter 7

DateTime Functions

Work with dates and times — get the current date, add intervals, find differences, and extract parts.

SQL
datetime_functions.sql
-- Current date / time
SELECT
    CURRENT_DATE,                             -- Standard SQL
    GETDATE(),                                -- SQL Server
    NOW(),                                     -- MySQL / PostgreSQL
    SYSDATE,                                  -- Oracle
    DATE('now')                               -- SQLite
FROM dual;

-- Add / subtract intervals
SELECT
    DATEADD(day, 30, order_date)            AS due_date,     -- SQL Server
    order_date + INTERVAL '30' day           AS due_date      -- PostgreSQL
FROM orders;

-- Difference between dates
SELECT
    DATEDIFF(day, order_date, GETDATE())   AS days_old      -- SQL Server
FROM orders;

-- Extract parts of a date
SELECT
    YEAR(order_date)         AS yr,
    MONTH(order_date)        AS mo,
    DAY(order_date)          AS dy,
    EXTRACT(MONTH FROM order_date) AS mo2    -- Standard SQL
FROM orders;
Chapter 7

NULL Functions

Handle missing data gracefully. NULL is not zero or empty string — it means unknown.

SQL
null_handling.sql
SELECT
    -- Return fallback if NULL
    COALESCE(payment_amount, 0)          AS amount,  -- all RDBMS
    ISNULL(payment_amount, 0)           AS amount,  -- SQL Server
    IFNULL(payment_amount, 0)           AS amount,  -- MySQL / SQLite
    NVL(payment_amount, 0)              AS amount   -- Oracle
FROM payments;

-- NULLIF — return NULL when two values are equal
SELECT NULLIF(quantity, 0) -- avoids divide-by-zero
FROM inventory;

-- Safe division pattern
SELECT revenue / NULLIF(units_sold, 0) AS price_per_unit
FROM sales;
Chapter 8

CASE Statements

If-then logic inside SQL queries. Use in SELECT, WHERE, ORDER BY, or GROUP BY.

SQL
case_statements.sql
-- Simple CASE: compare one column to values
SELECT
    order_id,
    CASE status
        WHEN 'paid'      THEN '✓ Paid'
        WHEN 'pending'   THEN '⏳ Pending'
        WHEN 'cancelled' THEN '✗ Cancelled'
        ELSE                   'Unknown'
    END AS status_label
FROM orders;

-- Searched CASE: evaluate conditions
SELECT
    order_id,
    order_total,
    CASE
        WHEN order_total >= 1000 THEN 'High Value'
        WHEN order_total >= 500  THEN 'Mid Value'
        ELSE                          'Low Value'
    END AS order_tier
FROM orders;

-- CASE inside aggregate (conditional count)
SELECT
    retailer,
    COUNT(CASE WHEN status = 'paid'    THEN 1 END) AS paid_orders,
    COUNT(CASE WHEN status = 'pending' THEN 1 END) AS pending_orders
FROM orders
GROUP BY retailer;
Chapter 9

JOINs

Combine rows from two or more tables based on a related column. The most powerful concept in relational databases.

JOIN Types

SQL
joins.sql
-- INNER JOIN: only rows that match in both tables
SELECT
    o.order_id,
    o.order_total,
    p.payment_amount
FROM       orders   o
INNER JOIN payments p  ON o.order_id = p.order_id;

-- LEFT JOIN: all orders, matched payment or NULL
SELECT
    o.order_id,
    o.order_total,
    p.payment_amount             -- NULL if no payment found
FROM      orders   o
LEFT JOIN payments p  ON o.order_id = p.order_id;

-- Find unmatched orders (no payment)
SELECT o.order_id, o.order_total
FROM      orders   o
LEFT JOIN payments p  ON o.order_id = p.order_id
WHERE p.order_id IS NULL;

-- FULL OUTER JOIN: all rows from both tables
SELECT o.order_id, p.payment_id
FROM            orders   o
FULL OUTER JOIN payments p  ON o.order_id = p.order_id;

-- Join 3 tables
SELECT
    o.order_id,
    c.customer_name,
    p.product_name
FROM       orders    o
INNER JOIN customers c  ON o.customer_id = c.customer_id
INNER JOIN products  p  ON o.product_id  = p.product_id;
Chapter 8

Window Functions

Perform calculations across a sliding window of rows without collapsing them into groups. Uses the OVER() clause.

Syntax Pattern function_name() OVER (PARTITION BY col ORDER BY col ROWS/RANGE BETWEEN ...)

Ranking Functions

SQL
window_rank.sql
SELECT
    retailer,
    order_id,
    order_total,

    -- Rank within each retailer, gaps on ties
    RANK() OVER (
        PARTITION BY retailer
        ORDER BY order_total DESC
    ) AS rank_in_retailer,

    -- Row number (no gaps)
    ROW_NUMBER() OVER (
        PARTITION BY retailer
        ORDER BY order_total DESC
    ) AS row_num,

    -- Dense rank (no gaps on ties)
    DENSE_RANK() OVER (
        PARTITION BY retailer
        ORDER BY order_total DESC
    ) AS dense_rank

FROM orders;

Running Total & Moving Average

SQL
window_running.sql
SELECT
    order_date,
    order_total,

    -- Running total of all orders so far
    SUM(order_total) OVER (
        ORDER BY order_date
        ROWS BETWEEN UNBOUNDED PRECEDING AND CURRENT ROW
    ) AS running_total,

    -- 3-day moving average
    AVG(order_total) OVER (
        ORDER BY order_date
        ROWS BETWEEN 2 PRECEDING AND CURRENT ROW
    ) AS moving_avg_3d,

    -- Value from the previous row
    LAG(order_total, 1) OVER (ORDER BY order_date)  AS prev_total,

    -- First value in partition
    FIRST_VALUE(order_total) OVER (
        PARTITION BY retailer
        ORDER BY order_date
    ) AS first_order_total

FROM orders;
Chapter 9

CTEs & Subqueries

CTEs (Common Table Expressions) create named temporary result sets. Cleaner than nested subqueries. Recursive CTEs can traverse hierarchical data.

Basic CTE

SQL
cte_basic.sql
-- CTE: WITH ... AS ()
WITH paid_orders AS (
    SELECT *
    FROM   orders
    WHERE  status = 'paid'
),
retailer_totals AS (
    SELECT
        retailer,
        SUM(order_total) AS total
    FROM   paid_orders
    GROUP BY retailer
)
SELECT *
FROM   retailer_totals
WHERE  total > 50000
ORDER BY total DESC;

Recursive CTE

SQL
cte_recursive.sql
-- Traverse org hierarchy: employee → manager chain
WITH RECURSIVE org_tree AS (
    -- Anchor: start from top-level (no manager)
    SELECT emp_id, emp_name, manager_id, 0 AS level
    FROM   employees
    WHERE  manager_id IS NULL

    UNION ALL

    -- Recursive step: add each employee under their manager
    SELECT e.emp_id, e.emp_name, e.manager_id, t.level + 1
    FROM   employees e
    JOIN   org_tree   t  ON e.manager_id = t.emp_id
)
SELECT * FROM org_tree
ORDER BY level, emp_name;
Chapter 9

UNION / EXCEPT / INTERSECT

Combine or compare result sets from multiple queries. Columns must match in number and data type.

SQL
set_operators.sql
-- UNION: combine, remove duplicates
SELECT customer_id FROM qvc_orders
UNION
SELECT customer_id FROM hsn_orders;

-- UNION ALL: combine, keep duplicates (faster)
SELECT order_id, order_total, 'QVC' AS source
FROM qvc_orders
UNION ALL
SELECT order_id, order_total, 'HSN' AS source
FROM hsn_orders;

-- EXCEPT: rows in first query but not second
SELECT order_id FROM invoices
EXCEPT
SELECT order_id FROM payments;  -- invoiced but never paid

-- INTERSECT: rows present in both queries
SELECT order_id FROM invoices
INTERSECT
SELECT order_id FROM payments;  -- invoiced AND paid
Chapter 10

How Do I…?

Practical solutions to the most frequently searched SQL problems.

Find Duplicate Rows

SQL
find_duplicates.sql
-- Find duplicate invoice numbers
SELECT
    invoice_num,
    COUNT(*) AS occurrences
FROM invoices
GROUP BY invoice_num
HAVING COUNT(*) > 1
ORDER BY occurrences DESC;

-- Show all duplicate rows in full
SELECT * FROM invoices
WHERE invoice_num IN (
    SELECT invoice_num
    FROM   invoices
    GROUP BY invoice_num
    HAVING COUNT(*) > 1
);

Select Row with MAX Value per Group

SQL
max_per_group.sql
-- Latest order per retailer — using ROW_NUMBER()
WITH ranked AS (
    SELECT
        *,
        ROW_NUMBER() OVER (
            PARTITION BY retailer
            ORDER BY order_date DESC
        ) AS rn
    FROM orders
)
SELECT *
FROM   ranked
WHERE  rn = 1;

Find All Tables with a Specific Column

SQL
find_column.sql
-- SQL Server / MySQL: search INFORMATION_SCHEMA
SELECT
    TABLE_NAME,
    COLUMN_NAME,
    DATA_TYPE
FROM   INFORMATION_SCHEMA.COLUMNS
WHERE  COLUMN_NAME LIKE '%order_id%'
ORDER BY TABLE_NAME;

Update Where ID Matches Another Table

SQL
update_join.sql
-- SQL Server: UPDATE with JOIN
UPDATE o
SET    o.status = 'reconciled'
FROM       orders   o
INNER JOIN payments p  ON o.order_id = p.order_id
WHERE p.payment_amount >= o.order_total;

-- MySQL: UPDATE with subquery
UPDATE orders
SET    status = 'reconciled'
WHERE  order_id IN (
    SELECT order_id
    FROM   payments
    WHERE  payment_amount >= order_total
);

Concatenate Text from Multiple Columns

SQL
concat.sql
-- Standard SQL / PostgreSQL / SQLite
SELECT
    first_name || ' ' || last_name          AS full_name
FROM customers;

-- MySQL / SQL Server
SELECT
    CONCAT(first_name, ' ', last_name)    AS full_name
FROM customers;

-- Aggregate rows into comma-separated list
SELECT
    retailer,
    GROUP_CONCAT(product_name SEPARATOR ', ') AS products  -- MySQL
FROM order_items
GROUP BY retailer;

-- PostgreSQL equivalent
SELECT
    retailer,
    STRING_AGG(product_name, ', ') AS products
FROM order_items
GROUP BY retailer;
